class Solution {
    public String longestPalindrome(String s) {
        // 把字符串s变成一个字符数组
        char[] newS = s.toCharArray();
        int n = newS.length;
        int begin = 0;
        int end = 0;

        // 奇字符串
        for(int i = 0; i < n; i++){
            int l = i, r = i;
            while(l >= 0 && r < n && newS[l] == newS[r]){
                l--;
                r++;
            }
            // 此时newS[l+1]到newS[r-1]是回文串

            if (r - l - 1 > end - begin){
                begin = l + 1;
                end = r;      // 是一个左开右闭区间, 因为最后使用substring方法得到子字符串, 它就是包含左边界不包含右边界
            }
        }

        // 偶字符串
        for(int i = 0; i < n - 1; i++){
            int l = i, r = i + 1;
            while(l >= 0 && r < n && newS[l] == newS[r]){
                l--;
                r++;
            }
            if (r - l - 1 > end - begin){
                begin = l + 1;
                end = r;
            }
        }

        return s.substring(begin, end);
    }
}
